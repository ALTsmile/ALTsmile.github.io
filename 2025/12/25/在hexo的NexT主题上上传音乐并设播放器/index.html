在hexo的NexT主题上上传音乐并设播放器

发表于 2025-12-25 更新于 2025-12-27 本文字数： 24k 阅读时长 ≈ 22 分钟

本来想用网易云外链简单明了的，结果发现有些歌曲网易云要vip，我就想着能不能上传本地音乐到仓库，然后让AI写个简单的播放器功能

1. 下载Aplayer插件并配置

  • 下载Aplayer插件
npm install hexo-tag-aplayer --save
  • 然后在NexT的配置文件_config.yml中在最后一行输入以下配置信息
aplayer:
  cdn: https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.js
  style_cdn: https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.css
  meting: true # 如果要使用 MetingJS

2. 配置一个音乐播放器

这一步比较麻烦

  • 配置NexT的_config.yml

我选择把播放器寄存在侧边栏，让侧边栏调用。所以找到custom_file_path，把sidebar前的#去掉，然后写上路径source/_data/sidebar-music.njk

  • 创建存放歌曲和歌曲封面的文件夹

这一步我选择把歌曲和歌曲封面分别存放到source/music和source/music-cover

  • 配置音乐播放器

在source/_data下新建一个文件sidebar-music.njk，然后找AI生成代码，我这边已经调试了很久，把代码复制粘贴上去即可

function initMusicPlayer() {
  // 创建隐藏的 APlayer 容器
  const playerContainer = document.createElement('div');
  playerContainer.id = 'aplayer-hidden-container';
  playerContainer.style.display = 'none';
  document.body.appendChild(playerContainer);
  
  // 获取基础路径
  const basePath = getBasePath();
  
  // 修正音乐文件路径
  const correctedMusicList = musicList.map(track => {
    return {
      ...track,
      url: basePath + track.url,
      cover: basePath + track.cover
    };
  });
  
  // 简化：始终使用列表循环
  ap = new APlayer({
    container: document.getElementById('aplayer-hidden-container'),
    audio: correctedMusicList,
    theme: '#667eea',
    loop: 'all', // 始终使用全部循环
    order: 'list',
    preload: 'auto',
    volume: currentVolume,
    mutex: true,
    lrcType: 0,
    fixed: false,
    mini: false,
    autoplay: false,
    controls: false
  });
  
  // 事件监听
  ap.on('play', () => {
    isPlaying = true;
    updatePlayButton();
    updateAvatarPlayButton();
  });

  ap.on('pause', () => {
    isPlaying = false;
    updatePlayButton();
    updateAvatarPlayButton();
  });

  ap.on('canplay', () => {
    updateTrackInfo();
  });
  
  // 监听 timeupdate 更新进度
  ap.on('timeupdate', () => {
    if (!isSeeking && !isDraggingProgress) {
      updateProgress();
    }
  });
  
  ap.on('loadedmetadata', () => {
    updateProgress();
    const totalTime = document.getElementById('total-time');
    if (ap.audio.duration && !isNaN(ap.audio.duration)) {
      totalTime.textContent = formatTime(ap.audio.duration);
    }
  });
  
  // 监听歌曲切换
  ap.on('listswitch', (index) => {
    currentTrackIndex = index.item;
    updateTrackInfo();
    updateActivePlaylistItem(index.item);
  });
  
  // 初始化 UI
  renderPlaylist();
  setupEventListeners();
  positionPanel();
  updatePlayModeUI();
  hidePanel();
}

function setupEventListeners() {
  const avatar = document.getElementById('music-avatar');
  const panel = document.getElementById('music-panel');
  const progressBar = document.getElementById('progress-bar');
  const playlistItems = document.getElementById('playlist');
  const avatarBtn = document.getElementById('avatar-play-btn');
  const modeBtn = document.getElementById('mode-btn');
  const prevBtn = document.getElementById('prev-btn');
  const mainBtn = document.getElementById('main-play-btn');
  const nextBtn = document.getElementById('next-btn');

  if (!avatar || !panel || !progressBar) {
    console.error('播放器元素未找到');
    return;
  }

  // 滚动传递问题
  playlistItems.addEventListener('wheel', function(e) {
    const isAtBottom = this.scrollHeight - this.scrollTop <= this.clientHeight + 1;
    const isAtTop = this.scrollTop === 0;
    
    if (isAtBottom && e.deltaY > 0) {
      e.stopPropagation();
      e.preventDefault();
      return false;
    }
    
    if (isAtTop && e.deltaY < 0) {
      e.stopPropagation();
      e.preventDefault();
      return false;
    }
    
    e.stopPropagation();
  }, { passive: false });

  // 封面鼠标事件
  avatar.addEventListener('mouseenter', function(e) {
    e.stopPropagation();
    showPanel();
  });

  avatar.addEventListener('mouseleave', function(e) {
    e.stopPropagation();
    if (!isPanelVisible) {
      hidePanel();
    }
  });

  // 进度条事件处理
  progressBar.addEventListener('mousedown', function(e) {
    e.preventDefault();
    e.stopPropagation();
    
    if (!ap.audio || !ap.audio.duration) return;
    
    isDraggingProgress = true;
    progressBar.classList.add('dragging');
    isSeeking = true;
    
    const rect = progressBar.getBoundingClientRect();
    const x = e.clientX;
    const percent = Math.max(0, Math.min(1, (x - rect.left) / rect.width));
    
    updateProgressDisplay(percent);
    
    document.addEventListener('mousemove', handleProgressDragMove);
    document.addEventListener('mouseup', handleProgressDragEnd);
  });

  progressBar.addEventListener('click', function(e) {
    e.stopPropagation();
    
    if (!ap.audio || !ap.audio.duration) return;
    
    const rect = progressBar.getBoundingClientRect();
    const x = e.clientX;
    const percent = Math.max(0, Math.min(1, (x - rect.left) / rect.width));
    const targetTime = percent * ap.audio.duration;
    
    isSeeking = true;
    ap.seek(targetTime);
    updateProgressDisplay(percent);
    
    if (isPlaying) {
      setTimeout(() => {
        ap.play();
      }, 10);
    }
    
    setTimeout(() => {
      isSeeking = false;
    }, 100);
  });

  // 按钮事件
  avatarBtn.addEventListener('click', function(e) {
    e.stopPropagation();
    togglePlayPause();
  });

  modeBtn.addEventListener('click', function(e) {
    e.stopPropagation();
    changePlayMode();
  });

  prevBtn.addEventListener('click', function(e) {
    e.stopPropagation();
    previousTrack();
  });

  mainBtn.addEventListener('click', function(e) {
    e.stopPropagation();
    togglePlayPause();
  });

  nextBtn.addEventListener('click', function(e) {
    e.stopPropagation();
    nextTrack();
  });

  // 面板鼠标事件
  panel.addEventListener('mouseenter', function(e) {
    e.stopPropagation();
    isPanelVisible = true;
    if (panelTimeout) {
      clearTimeout(panelTimeout);
    }
  });

  panel.addEventListener('mouseleave', function(e) {
    e.stopPropagation();
    isPanelVisible = false;
    panelTimeout = setTimeout(hidePanel, 300);
  });

  window.addEventListener('scroll', positionPanel);
  window.addEventListener('resize', positionPanel);
  
  document.addEventListener('click', function(e) {
    const avatar = document.getElementById('music-avatar');
    const panel = document.getElementById('music-panel');
    if (!avatar.contains(e.target) && !panel.contains(e.target)) {
      hidePanel();
    }
  });
}

// 进度条拖拽移动
function handleProgressDragMove(e) {
  if (!isDraggingProgress || !ap.audio || !ap.audio.duration) return;
  
  e.preventDefault();
  e.stopPropagation();
  
  const progressBar = document.getElementById('progress-bar');
  const rect = progressBar.getBoundingClientRect();
  const x = e.clientX;
  const percent = Math.max(0, Math.min(1, (x - rect.left) / rect.width));
  
  updateProgressDisplay(percent);
}

// 进度条拖拽结束
function handleProgressDragEnd(e) {
  if (!isDraggingProgress || !ap.audio || !ap.audio.duration) return;
  
  e.preventDefault();
  e.stopPropagation();
  
  isDraggingProgress = false;
  const progressBar = document.getElementById('progress-bar');
  progressBar.classList.remove('dragging');
  
  const rect = progressBar.getBoundingClientRect();
  const x = e.clientX;
  const percent = Math.max(0, Math.min(1, (x - rect.left) / rect.width));
  const targetTime = percent * ap.audio.duration;
  
  ap.seek(targetTime);
  
  if (isPlaying) {
    ap.pause();
    setTimeout(() => {
      ap.play();
    }, 10);
  }
  
  updateProgressDisplay(percent);
  
  document.removeEventListener('mousemove', handleProgressDragMove);
  document.removeEventListener('mouseup', handleProgressDragEnd);
  
  setTimeout(() => {
    isSeeking = false;
  }, 100);
}

function updateProgressDisplay(percent) {
  const progress = document.getElementById('progress');
  const currentTimeEl = document.getElementById('current-time');
  
  if (progress) {
    progress.style.width = `${percent * 100}%`;
  }
  
  if (currentTimeEl && ap.audio && ap.audio.duration) {
    const targetTime = percent * ap.audio.duration;
    currentTimeEl.textContent = formatTime(targetTime);
  }
}

function updateProgress() {
  if (!ap.audio || !ap.audio.duration || isNaN(ap.audio.duration) || isSeeking || isDraggingProgress) {
    return;
  }
  
  const currentTime = ap.audio.currentTime || 0;
  const duration = ap.audio.duration || 0;
  const percent = duration > 0 ? (currentTime / duration) * 100 : 0;
  
  const progress = document.getElementById('progress');
  if (progress) {
    progress.style.width = `${percent}%`;
  }
  
  const currentTimeEl = document.getElementById('current-time');
  if (currentTimeEl) currentTimeEl.textContent = formatTime(currentTime);
}

function showPanel() {
  const panel = document.getElementById('music-panel');
  if (panel) {
    panel.classList.add('show');
    positionPanel();
    isPanelVisible = true;
    
    if (panelTimeout) {
      clearTimeout(panelTimeout);
    }
  }
}

function hidePanel() {
  const panel = document.getElementById('music-panel');
  if (panel) {
    panel.classList.remove('show');
    isPanelVisible = false;
  }
}

function positionPanel() {
  const avatar = document.getElementById('music-avatar');
  const panel = document.getElementById('music-panel');
  if (!avatar || !panel) return;
  
  const avatarRect = avatar.getBoundingClientRect();
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  const panelWidth = 280;
  const panelHeight = 320;
  
  let left = avatarRect.right + 10;
  let top = avatarRect.top;
  
  if (left + panelWidth > viewportWidth - 20) {
    left = avatarRect.left - panelWidth - 10;
  }
  
  if (left < 20) {
    left = 20;
  }
  
  if (top + panelHeight > viewportHeight - 20) {
    top = viewportHeight - panelHeight - 20;
  }
  
  if (top < 20) {
    top = 20;
  }
  
  panel.style.left = left + 'px';
  panel.style.top = top + 'px';
}

function renderPlaylist() {
  const playlistEl = document.getElementById('playlist');
  if (!playlistEl) return;
  
  playlistEl.innerHTML = '';
  
  const basePath = getBasePath();
  const correctedMusicList = musicList.map(track => ({
    ...track,
    url: basePath + track.url,
    cover: basePath + track.cover
  }));
  
  correctedMusicList.forEach((track, index) => {
    const item = document.createElement('div');
    item.className = 'playlist-item';
    if (index === 0) item.classList.add('active');
    item.innerHTML = `
      <div class="playlist-num">${index + 1}</div>
      <div class="playlist-info">
        <div class="playlist-track">${track.name}</div>
        <div class="playlist-artist">${track.artist}</div>
      </div>
      <div class="playlist-duration">${track.duration}</div>
    `;
    item.addEventListener('click', (e) => {
      e.stopPropagation();
      playTrack(index);
    });
    playlistEl.appendChild(item);
  });
}

function playTrack(index) {
  currentTrackIndex = index;
  
  if (playMode === 'random') {
    lastRandomIndex = index;
  }
  
  ap.list.switch(index);
  ap.play();
  updateActivePlaylistItem(index);
  updateTrackInfo();
  showPanel();
}

function updateActivePlaylistItem(index) {
  document.querySelectorAll('.playlist-item').forEach((item, i) => {
    item.classList.toggle('active', i === index);
  });
}

function previousTrack() {
  if (playMode === 'random') {
    let randomIndex;
    do {
      randomIndex = Math.floor(Math.random() * musicList.length);
    } while (musicList.length > 1 && randomIndex === lastRandomIndex);
    
    lastRandomIndex = randomIndex;
    playTrack(randomIndex);
  } else {
    ap.skipBack();
    if (!isPlaying) ap.play();
  }
  updateTrackInfo();
  showPanel();
}

function nextTrack() {
  if (playMode === 'random') {
    let randomIndex;
    do {
      randomIndex = Math.floor(Math.random() * musicList.length);
    } while (musicList.length > 1 && randomIndex === lastRandomIndex);
    
    lastRandomIndex = randomIndex;
    playTrack(randomIndex);
  } else {
    ap.skipForward();
    if (!isPlaying) ap.play();
  }
  updateTrackInfo();
  showPanel();
}

function changePlayMode() {
  // 只有两种模式：列表循环和随机播放
  const modes = [
    { id: 'list', name: '列表循环', icon: 'fa-list-ol' },
    { id: 'random', name: '随机播放', icon: 'fa-random' }
  ];
  
  const currentIndex = modes.findIndex(mode => mode.id === playMode);
  const nextIndex = (currentIndex + 1) % modes.length;
  playMode = modes[nextIndex].id;
  
  if (playMode === 'random') {
    // 初始化随机播放
    lastRandomIndex = -1;
  }
  
  updatePlayModeUI();
  showPanel();
}

function updatePlayModeUI() {
  const modeBtn = document.getElementById('mode-btn');
  const modeIcon = document.getElementById('mode-icon');
  const indicator = document.getElementById('play-mode-indicator');
  
  let title = '', icon = '', indicatorClass = '';
  
  if (playMode === 'list') {
    title = '列表循环';
    icon = 'fa-list-ol';
    indicatorClass = 'list';
  } else if (playMode === 'random') {
    title = '随机播放';
    icon = 'fa-random';
    indicatorClass = 'random';
  }
  
  if (modeBtn) modeBtn.title = title;
  if (modeIcon) modeIcon.className = 'fa ' + icon;
  if (indicator) indicator.className = 'play-mode-indicator ' + indicatorClass;
}

function updateTrackInfo() {
  const basePath = getBasePath();
  const correctedMusicList = musicList.map(track => ({
    ...track,
    url: basePath + track.url,
    cover: basePath + track.cover
  }));
  
  const current = correctedMusicList[ap.list.index];
  if (!current) return;
  
  const trackName = document.getElementById('current-track-name');
  const trackArtist = document.getElementById('current-track-artist');
  const avatarImg = document.getElementById('avatar-img');
  const totalTime = document.getElementById('total-time');
  
  if (trackName) trackName.textContent = current.name;
  if (trackArtist) trackArtist.textContent = current.artist;
  if (avatarImg) {
    avatarImg.src = current.cover;
  }
  if (totalTime) totalTime.textContent = current.duration;
  
  updateActivePlaylistItem(ap.list.index);
}

function formatTime(seconds) {
  if (isNaN(seconds) || seconds === 0) return '00:00';
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
}
